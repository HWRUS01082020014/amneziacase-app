    <script>
        const tg = window.Telegram.WebApp;
        tg.expand();
        tg.MainButton.hide();
        
        let currentUser = null;
        let availableGifts = [];
        
        // Инициализация TON Connect
        let tonConnectUI = null;
        
        function initTonConnect() {
            if (typeof window.TonConnectUI !== 'undefined') {
                tonConnectUI = new window.TonConnectUI({
                    manifestUrl: 'https://your-domain.com/tonconnect-manifest.json',
                    buttonRootId: 'ton-connect-button'
                });
                
                tonConnectUI.onStatusChange((wallet) => {
                    if (wallet) {
                        const address = wallet.account.address;
                        sendToBot('connect_ton_wallet', { wallet_address: address });
                        tg.showPopup({
                            title: '✅ Кошелек подключен',
                            message: `Адрес: ${address.slice(0, 6)}...${address.slice(-4)}`,
                            buttons: [{ type: 'ok' }]
                        });
                    }
                });
            }
        }
        
        function sendToBot(action, data = {}) {
            const payload = JSON.stringify({ action: action, ...data });
            tg.sendData(payload);
        }
        
        function showLoading(message = 'Загрузка...') {
            document.getElementById('resultArea').innerHTML = `
                <div class="loading">
                    <div class="spinner"></div>
                    ${message}
                </div>
            `;
        }
        
        function hideLoading() {
            // Очищается при следующем действии
        }
        
        function showResult(title, message, isError = false) {
            document.getElementById('resultArea').innerHTML = `
                <div class="result-message" style="background: ${isError ? '#ffebee' : '#e8f5e9'}">
                    <strong>${title}</strong><br>
                    ${message}
                </div>
            `;
            
            // Авто-очистка через 5 секунд
            setTimeout(() => {
                const resultArea = document.getElementById('resultArea');
                if (resultArea.innerHTML.includes(title)) {
                    resultArea.innerHTML = '';
                }
            }, 5000);
        }
        
        function updateBalanceUI(balance) {
            document.getElementById('balance').innerText = balance;
        }
        
        function showGiftSelection(gifts) {
            availableGifts = gifts;
            const giftListDiv = document.getElementById('giftList');
            giftListDiv.innerHTML = '';
            
            gifts.forEach((gift, index) => {
                const giftDiv = document.createElement('div');
                giftDiv.className = 'gift-item';
                giftDiv.onclick = () => selectGift(index);
                giftDiv.innerHTML = `
                    <div class="gift-name">${gift.name} | ${gift.model} | ${gift.backdrop}</div>
                    <div class="gift-price">💰 ${gift.price} TON</div>
                `;
                giftListDiv.appendChild(giftDiv);
            });
            
            document.getElementById('giftModal').style.display = 'flex';
        }
        
        function closeGiftModal() {
            document.getElementById('giftModal').style.display = 'none';
        }
        
        function selectGift(index) {
            const gift = availableGifts[index];
            closeGiftModal();
            
            showLoading('Покупаем подарок...');
            sendToBot('open_case', {
                gift_type: gift.name,
                gift_price: gift.price
            });
        }
        
        // Загрузка баланса при старте
        function loadBalance() {
            sendToBot('get_balance');
        }
        
        // Загрузка статистики
        function loadStats() {
            showLoading('Загрузка статистики...');
            sendToBot('get_stats');
        }
        
        // Обработка ответов от бота
        window.handleBotResponse = function(response) {
            try {
                const data = JSON.parse(response);
                
                switch(data.type) {
                    case 'balance':
                        updateBalanceUI(data.balance);
                        document.getElementById('casePrice').innerText = data.case_price;
                        break;
                        
                    case 'stats':
                        hideLoading();
                        tg.showPopup({
                            title: '📊 Твоя статистика',
                            message: `💰 Баланс: ${data.balance} монет\n🎁 Открыто кейсов: ${data.cases_opened}\n💎 Пополнено: ${data.total_deposited} монет\n👛 TON кошелек: ${data.ton_wallet ? '✅ привязан' : '❌ не привязан'}`,
                            buttons: [{ type: 'ok' }]
                        });
                        break;
                        
                    case 'case_result':
                        if (data.success) {
                            updateBalanceUI(data.new_balance);
                            showResult(
                                '🎉 Поздравляем!',
                                `Ты получил: ${data.gift_name}\n🎨 Модель: ${data.gift_model}\n🖼️ Фон: ${data.gift_backdrop}\n💰 Цена: ${data.gift_price} TON\n\nПодарок уже в твоем профиле Telegram!`
                            );
                        } else {
                            showResult('❌ Ошибка', data.message, true);
                        }
                        break;
                        
                    case 'error':
                        hideLoading();
                        showResult('❌ Ошибка', data.message, true);
                        break;
                        
                    case 'wallet_connected':
                        if (data.success) {
                            showResult('✅ Кошелек подключен', `Адрес: ${data.wallet.slice(0, 6)}...${data.wallet.slice(-4)}`);
                        }
                        break;
                }
            } catch(e) {
                console.error('Ошибка обработки:', e);
            }
        };
        
        // Перехват сообщений от бота
        const originalSendData = tg.sendData;
        tg.sendData = function(data) {
            // Сохраняем оригинальное поведение
            originalSendData.call(tg, data);
        };
        
        // Обработка входящих данных (бот отвечает через message.answer)
        // Бот отправляет ответы как обычные сообщения, которые нужно парсить
        tg.onEvent('viewportChanged', () => {});
        
        // Обработчики кнопок
        document.getElementById('caseBtn').addEventListener('click', () => {
            // Здесь должен быть запрос к боту за списком доступных подарков
            // Для демонстрации показываем тестовые данные
            showGiftSelection([
                { name: 'dragon', model: 'Red Dragon', backdrop: 'Gold', price: 45 },
                { name: 'dragon', model: 'Blue Dragon', backdrop: 'Silver', price: 38 },
                { name: 'heroic helmet', model: 'Knight', backdrop: 'Gold', price: 52 }
            ]);
        });
        
        document.getElementById('topupStarsBtn').addEventListener('click', () => {
            tg.showPopup({
                title: '⭐ Пополнение Stars',
                message: 'Выбери сумму:',
                buttons: [
                    { id: '5', type: 'default', text: '⭐ 5 Stars → 50 монет' },
                    { id: '10', type: 'default', text: '⭐ 10 Stars → 100 монет' },
                    { id: '20', type: 'default', text: '⭐ 20 Stars → 200 монет' },
                    { id: 'cancel', type: 'cancel', text: 'Отмена' }
                ]
            }, (buttonId) => {
                if (buttonId !== 'cancel') {
                    sendToBot('topup_stars', { stars_amount: parseInt(buttonId) });
                }
            });
        });
        
        document.getElementById('topupTonBtn').addEventListener('click', () => {
            if (tonConnectUI) {
                tonConnectUI.openModal();
            } else {
                tg.showPopup({
                    title: '💎 Пополнение TON',
                    message: 'TON Connect пока настраивается.\n\nАльтернатива: отправь TON на кошелек бота и напиши /deposit в чат.',
                    buttons: [{ type: 'ok' }]
                });
            }
        });
        
        document.getElementById('statsBtn').addEventListener('click', loadStats);
        
        document.getElementById('connectWalletBtn').addEventListener('click', () => {
            if (tonConnectUI) {
                tonConnectUI.openModal();
            } else {
                tg.showPopup({
                    title: '👛 TON кошелек',
                    message: 'Чтобы привязать кошелек, установи Tonkeeper и нажми "Подключить"',
                    buttons: [{ type: 'ok' }]
                });
            }
        });
        
        // Инициализация
        initTonConnect();
        loadBalance();
        
        tg.ready();
    </script>
